## 1. 表情主题 IP（角色）表

```sql
CREATE TABLE `cloud_emoji_ip_info` (
  `ip_id` varchar(32) NOT NULL COMMENT '角色 IP 唯一标识',
  `user_id` varchar(32) NOT NULL COMMENT '所属用户 ID',
  `name` varchar(255) NOT NULL COMMENT '角色名称',
  `image_url` varchar(500) NOT NULL COMMENT '角色图片 OSS/CDN URL',
  `source_type` varchar(20) NOT NULL COMMENT '来源类型：ai_generated / user_upload',
  `source_keyword` varchar(255) DEFAULT NULL COMMENT '用户输入的角色关键词（AI生成时记录）',
  `prompt_text` text COMMENT 'AI扩写后的最终图像提示词',
  `description` varchar(1000) DEFAULT NULL COMMENT '角色描述（用户补充）',
  `is_default` tinyint(1) NOT NULL DEFAULT '0' COMMENT '是否默认角色：0-否，1-是',
  `create_time` datetime NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
  `update_time` datetime NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '更新时间',
  `status` char(1) NOT NULL DEFAULT '1' COMMENT '状态：1-有效，0-无效',
  PRIMARY KEY (`ip_id`),
  KEY `idx_user_id` (`user_id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci COMMENT='表情主题 IP（角色）表';
```

## 2. 表情包系列表

```sql
CREATE TABLE `cloud_emoji_group` (
  `emoji_group_id` varchar(32) NOT NULL COMMENT '系列唯一标识',
  `user_id` varchar(32) NOT NULL COMMENT '所属用户 ID',
  `ip_id` varchar(32) NOT NULL COMMENT '关联的角色 IP ID',
  `series_keywords` varchar(255) NOT NULL COMMENT '用户输入的系列关键词（逗号分隔）',
  `emoji_type` varchar(8) NOT NULL COMMENT '生成类型：static(静态) / dynamic(动态)',
  `emoji_style` varchar(255) DEFAULT NULL COMMENT '表情风格（如“可爱”、“写实”）',
  `create_time` datetime NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
  `update_time` datetime NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '更新时间',
  `status` char(1) NOT NULL DEFAULT '1' COMMENT '状态：1-有效，0-无效',
  PRIMARY KEY (`emoji_group_id`),
  KEY `idx_user_id` (`user_id`),
  KEY `idx_ip_id` (`ip_id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci COMMENT='表情包系列表';
```

## 3. 表情包资产表

```sql
CREATE TABLE `cloud_emoji_info` (
  `emoji_id` varchar(32) NOT NULL COMMENT '表情包唯一标识',
  `user_id` varchar(32) NOT NULL COMMENT '所属用户 ID',
  `ip_id` varchar(32) NOT NULL COMMENT '关联的角色 IP ID',
  `emoji_group_id` varchar(32) NOT NULL COMMENT '所属系列 ID',
  `name` varchar(255) NOT NULL COMMENT '表情名称',
  `prompt_text` varchar(500) DEFAULT NULL COMMENT 'AI扩写后的最终图像提示词',
  `type` varchar(8) NOT NULL COMMENT '类型：static / dynamic',
  `file_url` varchar(500) NOT NULL COMMENT '文件 URL（PNG 或 GIF）',
  `file_size` int NOT NULL DEFAULT '0' COMMENT '文件大小（KB）',
  `create_time` datetime NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
  `update_time` datetime NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '更新时间',
  `status` char(1) NOT NULL DEFAULT '1' COMMENT '状态：1-有效，0-无效',
  PRIMARY KEY (`emoji_id`),
  KEY `idx_user_id` (`user_id`),
  KEY `idx_group_id` (`emoji_group_id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci COMMENT='表情包资产表';
```
